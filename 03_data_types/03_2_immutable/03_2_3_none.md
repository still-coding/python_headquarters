## Что такое None?

`None` — это специальное значение, обозначающее отсутствие значения.

```python
x = None
type(x)  # <class 'NoneType'>
```

## Проверка на None

```python
x = None

# Проверка на None
x is None      # True (проверка идентичности)
x is not None  # False
```

## Сравнение 🆚 Идентичность

В Python существует два типа сравнения объектов, которые выполняют разные функции:

### 📊 Сравнение значений `==`

Оператор `==` проверяет **равенство значений** двух объектов.

```python
5 == 5        # True
5 == 5.0      # True (разные типы, но одинаковое значение)
10 == 5 + 5   # True (выражения вычисляются)
```

### 🔗 Проверка идентичности `is`

Оператор `is` проверяет, ссылаются ли две переменные на **один и тот же объект** в памяти.

```python
# 🚥 Проверка на None (рекомендуемый способ)
x = None
x is None          # True
x is not None      # False

# ⚠️ Неочевидное поведение с числами
a = 256
b = 256
a is b             # True (из-за кэширования малых чисел)

c = 1000
d = 1000
c is d             # False (разные объекты в памяти)

# 📝 Работа со списками
list1 = [1, 2, 3]
list2 = [1, 2, 3]
list3 = list1

list1 == list2     # True (одинаковые значения)
list1 is list2     # False (разные объекты)
list1 is list3     # True (одна и та же ссылка)
```

## 🤔 Когда что использовать?

- Используй `==` для сравнения **значений**
- Используй `is` для проверки, является ли объект **тем же самым объектом**
- Всегда используй `is` для сравнения с `None`, `True` и `False`

```python
# ✅ Правильно
if x is None:
    print("x is None")

# ⛔ Неправильно (хотя часто работает)
if x == None:
    print("x is None")
```

## 💡 Полезное замечание

```python
# Один из лучших способов понять разницу:
a = [1, 2, 3]
b = a
c = [1, 2, 3]

a == b  # True (одинаковые значения)
a is b  # True (один и тот же объект)
a == c  # True (одинаковые значения)
a is c  # False (разные объекты)

id(a)  # Уникальный идентификатор объекта (в CPython — адрес в памяти)
id(b)  # Тот же идентификатор
id(c)  # Другой идентификатор
```

Если тебе нужно проверить равенство значений, используй `==`. Если тебе нужно узнать, является ли объект тем же самым объектом в памяти, используй `is`.

## Практические примеры использования None

### Функции без возвращаемого значения

```python
def print_greeting(name):
    """Функция, которая ничего не возвращает"""
    print(f"Привет, {name}!")
    # Неявно возвращает None

result = print_greeting("Иван")
print(result)  # None
print(result is None)  # True
```

### Инициализация переменных

```python
# Инициализация переменной без значения
user_name = None
user_age = None

# Позже присваиваем значения
user_name = "Иван"
user_age = 25

# Проверяем, были ли значения присвоены
if user_name is not None:
    print(f"Пользователь: {user_name}, возраст: {user_age}")
else:
    print("Данные пользователя не заданы")
```

### Обработка отсутствующих данных

```python
def get_user_data(user_id):
    """Получает данные пользователя по ID"""
    # Имитируем базу данных
    users = {
        1: {"name": "Иван", "age": 25},
        2: {"name": "Петр", "age": 30}
    }
    
    return users.get(user_id)  # Возвращает None, если пользователь не найден

# Использование
user = get_user_data(1)
if user is not None:
    print(f"Найден пользователь: {user['name']}")
else:
    print("Пользователь не найден")

user = get_user_data(999)
if user is not None:
    print(f"Найден пользователь: {user['name']}")
else:
    print("Пользователь не найден")  # Этот код выполнится
```

### Опциональные параметры

```python
def create_user(name, email, phone=None, address=None):
    """Создает пользователя с опциональными полями"""
    user = {
        'name': name,
        'email': email
    }
    
    # Добавляем опциональные поля только если они не None
    if phone is not None:
        user['phone'] = phone
    
    if address is not None:
        user['address'] = address
    
    return user

# Создаем пользователей
user1 = create_user("Иван", "ivan@example.com")
user2 = create_user("Петр", "petr@example.com", phone="+7-999-123-45-67")

print(user1)  # {'name': 'Иван', 'email': 'ivan@example.com'}
print(user2)  # {'name': 'Петр', 'email': 'petr@example.com', 'phone': '+7-999-123-45-67'}
```

### Обработка ошибок

```python
def safe_divide(a, b):
    """Безопасное деление с обработкой ошибок"""
    try:
        return a / b
    except ZeroDivisionError:
        return None
    except TypeError:
        return None

# Использование
result1 = safe_divide(10, 2)
result2 = safe_divide(10, 0)
result3 = safe_divide(10, "abc")

print(f"10 / 2 = {result1}")      # 10 / 2 = 5.0
print(f"10 / 0 = {result2}")      # 10 / 0 = None
print(f"10 / 'abc' = {result3}")  # 10 / 'abc' = None

# Проверяем результаты
if result1 is not None:
    print(f"Деление успешно: {result1}")
else:
    print("Ошибка при делении")
```

### Конфигурация со значениями по умолчанию

```python
def process_config(config):
    """Обрабатывает конфигурацию со значениями по умолчанию"""
    
    # Значения по умолчанию
    default_config = {
        'debug': False,
        'timeout': 30,
        'retries': 3
    }
    
    # Объединяем с переданной конфигурацией
    final_config = default_config.copy()
    
    if config is not None:
        final_config.update(config)
    
    return final_config

# Тестируем
config1 = None
config2 = {'debug': True, 'timeout': 60}
config3 = {'retries': 5}

print(process_config(config1))  # Все значения по умолчанию
print(process_config(config2))  # debug=True, timeout=60, retries=3
print(process_config(config3))  # debug=False, timeout=30, retries=5
```

### 🔧 Полезные паттерны

```python
# Паттерн "Значение или значение по умолчанию"
def get_config_value(key, default=None):
    """Получает значение конфигурации или возвращает значение по умолчанию"""
    config = {'timeout': 30, 'debug': False}
    return config.get(key, default)

# Паттерн "Проверка и обработка"
def process_data(data):
    """Обрабатывает данные, если они есть"""
    if data is None:
        print("Нет данных для обработки")
        return
    
    # Обработка данных
    print(f"Обрабатываем: {data}")

# Тестируем
print(get_config_value('timeout'))        # 30
print(get_config_value('missing', 100))   # 100

process_data("Тестовые данные")  # Обрабатываем: Тестовые данные
process_data(None)               # Нет данных для обработки
```
